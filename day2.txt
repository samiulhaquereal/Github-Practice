git log = Show history of all commit
git log --oneline = Show History in Short form
git reset --soft HEAD^ = Remove Last Commit but It will staging place
git reset --soft HEAD~2 = Remove Number of Commit but It will staging place
git reset HEAD^ = Remove Last Commit and also It remove staging place
git reset --hard HEAD^ = Go to last Commit & remove all thing
git show = Show last commit / Head all details
git checkout "commit id" = Back to Previous commit or Main Commit (master)
git remote add "name" "Link" = Add Local Rep to Github

git clone 'Link' = Clone Rep Github to Local
git branch -M main = Create Main branch for new Rep on Github ( Optional )
git push -u origin main = Push in Github
git pull origin main = Pull from Github

git branch = Show All Branch
git branch 'name' = Create New Branch
git checkout 'branch_name' = Switch Between Branch
git branch -b "branch_name" = Create brnach and Switch in New Branch
git branch -d 'branch_name' = Delete Branch

git merge 'branch_name' = Merge branch with main
